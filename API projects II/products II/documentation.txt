  try{
    const response = await fetch(url, {
        headers:{
            Accept: 'application/json',
            'User-Agent': 'learning app'
        }
    })
    if(!response.ok){
        throw new Error('There was an error')
    }
    const data = await response.json();
    log(data);
    result.textContent = data.joke;
  }catch(error){
    log(error.message);
    result.textContent = error.message;
  }
};

Typically you use async when you call a method that returns a Task, 
and need to schedule work after the Task completes. 
The easiest way to do that is to await the Task.
An async method is much slower (40x) compared to a normal sync method.
